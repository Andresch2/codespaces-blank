// @title: V6 progresivo dos tramos con tasas variables (COP)
// @expect: ok
// Unidades: COP (pesos colombianos)

// Construcción de ingreso con decimales y paréntesis
sueldo = (80000 + 20000) * 1.05;
extras = (12000 * 1.3) + (3500 / 1.4);
ingreso = sueldo + extras;

// Tasas por tramo
t1 = 0.18;   // 100k–200k
t2 = 0.22;   // >200k

impuesto = 0;

// Alto
si ingreso > 200000: impuesto = impuesto + (ingreso - 200000) * t2;

// Bajo si supera 200k (siempre 100k al 18%)
si ingreso > 200000: impuesto = impuesto + 100000 * t1;

// Bajo si está entre 100k y 200k
si ingreso <= 200000: impuesto = impuesto + (ingreso - 100000) * t1;

print(impuesto);
