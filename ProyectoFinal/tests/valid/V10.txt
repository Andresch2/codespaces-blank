// Unidades: COP (pesos colombianos)

// Partes del ingreso con ponderadores y divisor no entero
base    = 98765;
product = (43210 * 1.35) - (1250 / 2.5);
horas   = (37 * 4500) / 1.1;                // horas * tarifa / factor
bono1   = (base * 0.07);
bono2   = (product * 0.045);
ajuste  = (1 + (3/100) + (2/100));          // 5% acumulado
ingreso = (base + product + horas + bono1 + bono2) * ajuste;

// Tasas variables por tramo (t1, t2, t3)
t1 = (1/5) - (1/20);    // 0.20 - 0.05 = 0.15
t2 = 0.21;
t3 = (7/25) + (1/100);  // 0.28 + 0.01 = 0.29

// Impuesto acumulado
impuesto = 0;

// Tramo alto: > 300k
si ingreso > 300000: impuesto = impuesto + (ingreso - 300000) * t3;

// Tramo medio: 200kâ€“300k (si supera 200k, siempre 100k al t2)
si ingreso > 200000: impuesto = impuesto + 100000 * t2;

// Tramo bajo:
si ingreso > 200000: impuesto = impuesto + 100000 * t1;
si ingreso <= 200000: impuesto = impuesto + (ingreso - 100000) * t1;

print(impuesto);
